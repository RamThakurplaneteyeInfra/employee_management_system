================================================================================

================================================================================

--------------------------------------------------------------------------------
5. CREATE CLIENT LEAD
--------------------------------------------------------------------------------
POST /clientsapi/profiles/

Request:
{
  "company_name": "Acme Corp",
  "client_name": "John Doe",
  "client_contact": "9876543210",
  "representative_contact_number": "9876543211",
  "representative_name": "Jane Smith",
  "motive": "Describe the lead...",
  "gst_number": "27AABCU9603R1ZM",
  "status_id": 1,
  "product_name": "Cropeye",
  "members": [1, 2, 3]
}

Response 201:
{
  "id": 1,
  "message": "Client lead created"
}

Response 400:
{
  "error": "company_name and client_name are required"
}

--------------------------------------------------------------------------------
7. UPDATE CLIENT LEAD
--------------------------------------------------------------------------------
PUT /clientsapi/profiles/1/
PATCH /clientsapi/profiles/1/

Request:
{
  "company_name": "Acme Corp Updated",
  "client_name": "John Doe",
  "representative_name": "Jane Smith",
  "status_id": 2,
  "product_name": "RAMS",
  "members": [2, 3, 4]
}

Response 200:
{
  "message": "Client lead updated"
}

Response 404:
{
  "error": "Client not found"
}


--------------------------------------------------------------------------------
8. DELETE CLIENT LEAD
--------------------------------------------------------------------------------
DELETE /clientsapi/profiles/1/

Response 200:
{
  "message": "Client lead deleted"
}

Response 404:
{
  "error": "Client not found"
}


--------------------------------------------------------------------------------

]
